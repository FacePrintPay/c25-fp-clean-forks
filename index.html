echo -e "\033[0;34m‚ïî‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïó\033[0m"
echo -e "\033[0;34m‚ïë  FacePrintPay Fork Cleanup                 ‚ïë\033[0m"
echo -e "\033[0;34m‚ïö‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïù\033[0m"
echo -e "\033[0;36mScanning for forked repos...\033[0m"
# List all forks
gh repo list FacePrintPay --json name,isFork,owner -q '.[] | select(.isFork == true) | .name' > "$TEMP_DIR/forks.txt"
if [ ! -s "$TEMP_DIR/forks.txt" ]; then
    echo -e "\033[0;32m‚úÖ No forked repos found.\033[0m"
    rm -rf "$TEMP_DIR"
    exit 0
fi
echo -e "\033[0;33mFound forked repos:\033[0m"
cat "$TEMP_DIR/forks.txt"
echo -e "\033[0;33m\nType 'DELETE' to confirm and remove all forks:\033[0m"
read -r confirm
